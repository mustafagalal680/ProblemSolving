#include<iostream>
#include<string>
using namespace std;
 
int main() {
	/*
5 + 10 = 15
5 * 10 = 50
5 - 10 = -5
	*/
	long long a, b;
	cin >> a >> b;
	cout << a << " + " << b << " = " << a + b<<endl;
	cout << a << " * " << b << " = " << a * b<<endl;
	cout << a << " - " << b << " = " << a - b<<endl;
 
 
	return 0;
}
